// Use DBML to define your database structure
// Docs: https://dbml.dbdiagram.io/docs

Table AdminLogin {
  admin_id integer [primary key]
  admin_name varchar
  admin_username varchar
  admin_email varchar
  admin_password varchar
  created_at timestamp
  last_login timestamp
}

Table UserLogin {
  user_id integer [primary key]
  user_name varchar
  user_username varchar
  user_email varchar
  user_password varchar
  created_at timestamp
  last_login timestamp
}

Table UserAddress {
  user_id integer [ref: > UserLogin.user_id]
  address_line1 varchar
  address_line2 varchar
  postal_code integer
  city varchar
  state varchar
  country varchar
  mobile_number varchar
}

Table ProductCategory {
  category_id integer [primary key]
  category_name varchar
  category_description varchar
}

Table Product {
  product_id integer [primary key]
  category_id integer [ref: > ProductCategory.category_id]
  product_name varchar
  product_price varchar
  product_description varchar
  product_features varchar
  product_image_url varchar
}

Table ProductInventory {
  product_id integer [ref: > Product.product_id]
  product_quantity integer
  last_modified timestamp
}

Table Cart {
  cart_id integer [primary key]
  user_id integer [ref: > UserLogin.user_id]
  payment_id integer [ref: > Payment.payment_id]
  cart_code integer
  cart_total integer
  created_at timestamp
  last_modified timestamp
}

Table CartItems {
  cart_id integer [ref: > Cart.cart_id]
  item_id integer [ref: > Product.product_id]
  quantity integer
  created_at timestamp
}

Table Orders {
  order_id integer [primary key]
  cart_id integer [ref: > Cart.cart_id]
  user_id integer [ref: > UserLogin.user_id]
  order_status varchar
  placed_at timestamp
}

Table Return {
  return_id integer [primary key]
  order_id integer [ref: > Orders.order_id]
  user_id integer [ref: > UserLogin.user_id]
  bank_details varchar
  return_date timestamp
}

Table Payment {
  payment_id integer [primary key]
  order_id integer [ref: > Orders.order_id]
  payment_mode varchar 
  amount integer
  payment_time timestamp
}
